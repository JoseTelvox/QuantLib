35 class Tracing : public Singleton<Tracing> {

37 private:
38 Tracing(); // NOLINT(modernize-use-equals-delete)
39 public:
40 void enable() {
41 #if defined(QL_ENABLE_TRACING)
42 enabled_ = true;
43 #else
44 QL_FAIL("tracing support not available");
45 #endif
46 }
47 void disable() { enabled_ = false; }
48 void setStream(std::ostream& stream) { out_ = &stream; }
49 bool enabled() const { return enabled_; }
50 std::ostream& stream() { return *out_; }
51 Integer depth() const { return depth_; }
52 void down() { depth_++; }
53 void up() { depth_--; }
54 private:
55 std::ostream* out_;
56 bool enabled_ = false;
57 Integer depth_ = 0;
58 };
