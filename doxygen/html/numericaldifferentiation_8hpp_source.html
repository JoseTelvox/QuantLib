65 inline Real NumericalDifferentiation::operator()(Real x) const {
66 Real s = 0.0;
67 for (Size i=0; i < w_.size(); ++i) {
68 if (std::fabs(w_[i]) > QL_EPSILON*QL_EPSILON) {
69 s += w_[i] * f_(x+offsets_[i]);
70 }
71 }
72 return s;
73 }
