151 template <class Interpolator>
152 InterpolatedYoYInflationCurve<Interpolator>::InterpolatedYoYInflationCurve(
153 const Date& referenceDate,
154 std::vector<Date> dates,
155 const std::vector<Rate>& rates,
156 Frequency frequency,
157 const DayCounter& dayCounter,
158 const ext::shared_ptr<Seasonality>& seasonality,
159 const Interpolator& interpolator)
160 : YoYInflationTermStructure(referenceDate, dates.at(0), rates[0],
161 frequency, dayCounter, seasonality),
162 InterpolatedCurve<Interpolator>(std::vector<Time>(), rates, interpolator),
163 dates_(std::move(dates)) {
164
165 QL_REQUIRE(dates_.size()>1, "too few dates: " << dates_.size());
166
167 QL_REQUIRE(this->data_.size() == dates_.size(),
168 "indices/dates count mismatch: "
169 << this->data_.size() << " vs " << dates_.size());
170
171 for (Size i = 1; i < dates_.size(); i++) {
172 // YoY inflation data may be positive or negative
173 // but must be greater than -1
174 QL_REQUIRE(this->data_[i] > -1.0,
175 "year-on-year inflation data < -100 %");
176 }
177
178 this->setupTimes(dates_, referenceDate, dayCounter);
179 this->setupInterpolation();
180 this->interpolation_.update();
181 }

233 template <class Interpolator>
234 InterpolatedYoYInflationCurve<Interpolator>::InterpolatedYoYInflationCurve(
235 const Date& referenceDate,
236 const Calendar& calendar,
237 const DayCounter& dayCounter,
238 const Period& lag,
239 Frequency frequency,
240 bool indexIsInterpolated,
241 std::vector<Date> dates,
242 const std::vector<Rate>& rates,
243 const Interpolator& interpolator)
244 : YoYInflationTermStructure(
245 referenceDate, calendar, dayCounter, rates[0], lag, frequency, indexIsInterpolated),
246 InterpolatedCurve<Interpolator>(std::vector<Time>(), rates, interpolator),
247 dates_(std::move(dates)) {
248
249 QL_REQUIRE(dates_.size()>1, "too few dates: " << dates_.size());
250
251 // check that the data starts from the beginning,
252 // i.e. referenceDate - lag, at least must be in the relevant
253 // period
254 std::pair<Date,Date> lim =
255 inflationPeriod(referenceDate - this->observationLag(), frequency);
256 QL_REQUIRE(lim.first <= dates_[0] && dates_[0] <= lim.second,
257 "first data date is not in base period, date: " << dates_[0]
258 << " not within [" << lim.first << "," << lim.second << "]");
259
260 QL_REQUIRE(this->data_.size() == dates_.size(),
261 "indices/dates count mismatch: "
262 << this->data_.size() << " vs " << dates_.size());
263
264 for (Size i = 1; i < dates_.size(); i++) {
265 // YoY inflation data may be positive or negative
266 // but must be greater than -1
267 QL_REQUIRE(this->data_[i] > -1.0,
268 "year-on-year inflation data < -100 %");
269 }
270
271 this->setupTimes(dates_, referenceDate, dayCounter);
272 this->setupInterpolation();
273 this->interpolation_.update();
274 }
