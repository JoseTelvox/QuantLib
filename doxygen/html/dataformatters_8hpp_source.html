157 template <typename T>
158 inline std::ostream& operator<<(std::ostream& out,
159 const power_of_two_holder<T>& holder) {
160 if (holder.n == Null<T>())
161 return out << "null";
162
163 T n = holder.n;
164 Integer power = 0;
165 if (n != 0) {
166 while (!(n & 1UL)) {
167 power++;
168 n >>= 1;
169 }
170 }
171 return out << n << "*2^" << power;
172 }
